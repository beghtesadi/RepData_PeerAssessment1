AveStep <- vector()
AveInt <- vector()
AveWeek <- vector()
for(i in unique(NewDF$interval))
{
  for(t in levels(NewDF$WeekDay))
  {
    AveStep <- append(AveStep , mean(NewDF$steps[NewDF$interval == i & NewDF$WeekDay== t]))

  }
